        ->merge($apartment->charges->pluck('period')->map->year)

        ->map(fn($m) => $charges[$m]->sum('amount') ?? 0)
        ->toArray();

        ->map(fn($m) => $payments[$m]->sum('amount') ?? 0)
        ->toArray();

        'apartment',
        'charges',
        'payments',
        'years',
        'year',

        'totalPayments',

        'labels',
        'chargesGraph',
        'paymentsGraph'
